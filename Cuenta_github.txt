ALTER PROCEDURE INSERTAR_ALERTAS 
	@ALERTNAME VARCHAR(25),
	@ALERTDESCRIPCION VARCHAR(50),
	@AREA INT,
	@FRECUENCIA INT 

	AS BEGIN 
	
	IF NOT EXISTS (
		SELECT * 
		FROM ALERTS
		WHERE AlertName = @ALERTNAME
		AND AlertDescription = @ALERTDESCRIPCION
		AND Area = @AREA
	)
	BEGIN
		INSERT INTO ALERTS (AlertName, AlertDescription, AlertActivo, Area, AlertTimes, Frecuencia)
			VALUES (@ALERTNAME, @ALERTDESCRIPCION, 1, @AREA, 3, @FRECUENCIA);

	END

END
